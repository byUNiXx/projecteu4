defend_the_american_colonies = {
	
	type = country
	
	category = MIL

	allow = {
		normal_or_historical_nations = yes
		tag = USA
		is_free_or_tributary_trigger = yes
		northeast_america_region = {
			owned_by = USA
		}
		northeast_america_region = {
			owned_by = GBR
		}
	}
	abort = {
		is_subject_other_than_tributary_trigger = yes
	}
	success = {
		NOT = { war_with = GBR }
		northeast_america_region = {
			type = all
			owned_by = USA
		}
	}
	chance = {
		factor = 3000
		modifier = {
			factor = 2
			mil = 4
		}
		modifier = {
			factor = 2
			ENG = { NOT = { mil = 2 } }
		}
	}
	effect = {
		add_army_tradition = 50
		add_stability = 2
	}
}
